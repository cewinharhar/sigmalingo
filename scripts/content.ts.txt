export type ChallengeType = "SELECT" | "ASSIST";

export type ChallengeOption = {
  text: string;
  correct: boolean;
};

export type ChallengeTemplate = {
  type: ChallengeType;
  question: string;
  order: number;
  options: ChallengeOption[];
};

export type Lesson = {
  id: number;
  title: string;
  challenges: ChallengeTemplate[];
  order: number;
};

export type Unit = {
  id: number;
  title: string;
  description: string;
  lessons: Lesson[];
  order: number;
};

export type CourseContent = {
  id: number;
  title: string;
  imageSrc: string;
  description: string;
  units: Unit[];
};
